SELECT 
    a.id_colaborador,
    b.nombre_display,  -- asumiendo que hay un campo de nombre en la tabla colaboradores
    b.area_empresa, 
    ARRAY_AGG(d.sector) AS sectores,
		b.url_image
FROM test_meta_mg.colaboradores a
INNER JOIN dim_nomina_coorporacion b 
    ON a.id_user = b.iddae 
LEFT JOIN test_meta_mg.test_colaboradores_sector c 
    ON a.id_colaborador = c.id_colaborador
LEFT JOIN test_meta_mg.sectores d 
    ON c.id_sector = d.id_sector
WHERE 
    b.area_empresa = {{sel_area.selectedOptionValue}} 
    AND a.cargo = {{sel_cargo.selectedOptionValue}} 
    AND a.estado = 1 
    AND a.fecha_baja IS NULL
GROUP BY 
    a.id_colaborador, b.nombre_display, b.area_empresa, b.url_image;
